  Main update handler function
 
function handleUpdate($bot, $update) {
     Get user ID from message or callback query
    $userId = null;
    $chatId = null;
    $messageId = null;
    $userData = [];
    
     Handle message updates
    if ($update-getMessage()) {
        $message = $update-getMessage();
        $userId = $message-getFrom()-getId();
        $chatId = $message-getChat()-getId();
        $userData = [
            'id' = $userId,
            'first_name' = $message-getFrom()-getFirstName()  'Unknown',
            'last_name' = $message-getFrom()-getLastName()  '',
            'username' = $message-getFrom()-getUsername()  '',
            'language_code' = $message-getFrom()-getLanguageCode()  'en'
        ];
        
         Handle text commands
        $text = $message-getText();
        if ($text) {
            handleCommand($bot, $chatId, $text, $userId, $userData);
        }
    }
    
     Handle callback queries
    if ($update-getCallbackQuery()) {
        $callbackQuery = $update-getCallbackQuery();
        $userId = $callbackQuery-getFrom()-getId();
        $chatId = $callbackQuery-getMessage()-getChat()-getId();
        $messageId = $callbackQuery-getMessage()-getMessageId();
        $callbackData = $callbackQuery-getData();
        
        handleCallback($bot, $chatId, $messageId, $callbackData, $userId);
    }
    
     Registerupdate user in database
    if ($userId && !empty($userData)) {
        registerUser($userId, $userData);
    }
}

  Register or update user in the JSON database
 
function registerUser($userId, $userData) {
    $users = loadUsers();
    $userId = (string) $userId;
    
    if (!isset($users[$userId])) {
         New user
        $users[$userId] = [
            'id' = $userId,
            'first_name' = $userData['first_name']  'Unknown',
            'last_name' = $userData['last_name']  '',
            'username' = $userData['username']  '',
            'language_code' = $userData['language_code']  'en',
            'registered_at' = date('Y-m-d His'),
            'last_active' = date('Y-m-d His'),
            'status' = 'active',
            'commands_used' = 0
        ];
    } else {
         Update existing user
        $users[$userId]['first_name'] = $userData['first_name']  $users[$userId]['first_name'];
        $users[$userId]['last_name'] = $userData['last_name']  $users[$userId]['last_name'];
        $users[$userId]['username'] = $userData['username']  $users[$userId]['username'];
        $users[$userId]['language_code'] = $userData['language_code']  $users[$userId]['language_code'];
        $users[$userId]['last_active'] = date('Y-m-d His');
        $users[$userId]['commands_used'] = ($users[$userId]['commands_used']  0) + 1;
    }
    
    saveUsers($users);
    return $users[$userId];
}


  Load users from JSON file
 
function loadUsers() {
    $file = __DIR__ . 'users.json';
    
     If file doesn't exist, create it with empty array
    if (!file_exists($file)) {
        saveUsers([]);
        return [];
    }
    
     Read file
    $content = file_get_contents($file);
    if ($content === false) {
        error_log(Failed to read users.json);
        return [];
    }
    
    $users = json_decode($content, true);
    if ($users === null && json_last_error() !== JSON_ERROR_NONE) {
        error_log(JSON decode error  . json_last_error_msg());
        return [];
    }
    
    return $users  [];
}


  Save users to JSON file
 
function saveUsers($users) {
    $file = __DIR__ . 'users.json';
    
     Ensure directory exists and is writable
    $dir = dirname($file);
    if (!is_dir($dir)) {
        mkdir($dir, 0755, true);
    }
    
     Write to file with proper permissions
    $json = json_encode($users, JSON_PRETTY_PRINT  JSON_UNESCAPED_UNICODE);
    if ($json === false) {
        error_log(Failed to encode users to JSON);
        return false;
    }
    
    $result = file_put_contents($file, $json, LOCK_EX);
    if ($result === false) {
        error_log(Failed to write users.json);
        return false;
    }
    
     Set file permissions
    chmod($file, 0666);
    
    return true;
}


  Get server uptime
 
function getUptime() {
    if (function_exists('exec')) {
        $uptime = @exec('uptime -p');
        if ($uptime) {
            return str_replace('up ', '', $uptime);
        }
    }
    return 'Unknown';
}


  Format bytes to human readable format
 
function formatBytes($bytes, $precision = 2) {
    $units = ['B', 'KB', 'MB', 'GB', 'TB'];
    $bytes = max($bytes, 0);
    $pow = floor(($bytes  log($bytes)  0)  log(1024));
    $pow = min($pow, count($units) - 1);
    $bytes = pow(1024, $pow);
    return round($bytes, $precision) . ' ' . $units[$pow];
}
